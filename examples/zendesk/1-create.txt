# THIS EXAMPLE DEMONSTRATES HOW TO LOAD ZENDESK DATA TO GOODDATA

# CREATE A NEW PROJECT
CreateProject(name="Zendesk Sample Project");

# STORE THE PROJECT ID TO FILE FOR THE FOLLOWING SCRIPT
StoreProject(fileName="examples/zendesk/pid");

# LOAD THE ZENDESK DATA
LoadCsv(csvDataFile="examples/zendesk/data.csv", configFile="examples/zendesk/config.xml");

# GENERATE THE ZENDESK MAQL
GenerateMaql(maqlFile="examples/zendesk/zendesk.maql");

# EXECUTE THE ZENDESK MAQL
ExecuteMaql(maqlFile="examples/zendesk/zendesk.maql");
